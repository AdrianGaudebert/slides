Events

  • EnterFrame
  • KeyDown
  • Click
  • Change
  • List of all events
Crafty.e('2D, DOM, Text')
      .bind('EnterFrame', function () {
        this.x += 1;
        var newText = 'Frame number ';
        newText += Crafty.frame();
        this.text(newText);
      });

Animations

A running Caveman

Crafty.e('2D, DOM, SpriteAnimation, caveman1')
      .attr({ x: 50, y: 50 })
      .animate('walk', 0, 0, 3)
      .animate('walk', 4, -1);

Keyboard

Crafty.e('2D, Keyboard')
      .bind('EnterFrame', function () {
        if (this.isDown('ARROW_RIGHT') ||
            this.isDown('D')) {
          this.x += 1;
        }
      });

Mouse

Crafty.e('2D, Mouse')
      .areaMap(
        [0, 0], [0, 34],
        [34, 34], [34, 0]
      )
      .bind('Click', function () {
        console.log('CLICKED!!!11!1');
      });
